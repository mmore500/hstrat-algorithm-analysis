\begin{algorithm}
\caption{Fixed Resolution Drop Ranks Enumeration}
\label{alg:fixed-resolution-algo-gen-drop-ranks}
\begin{algorithmic}[1]
    \Require{ $n \in \mathbb{N}_{\geq0}$ -- the number of strata deposited }
    \Require{ $r \in \mathbb{N}_{>0}$  -- the fixed resolution desired }
    \Ensure{ dropped strata, if any }
    \If{$n > 0$ and $n - 1 \bmod r > 0$}
      \State \textbf{yield} n - 1
    \EndIf
\end{algorithmic}
\end{algorithm}
