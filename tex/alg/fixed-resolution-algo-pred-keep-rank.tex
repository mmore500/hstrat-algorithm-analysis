\begin{algorithm}
\caption{Fixed Resolution Stratum Retention Predicate}
\label{alg:fixed-resolution-pred-keep-rank}
\begin{algorithmic}[1]
    \REQUIRE{$\texttt{n}$ -- the number of strata deposited}
    \REQUIRE{$\texttt{r}$ -- the fixed resolution desired}
    \FOR{$i = 0$ \textbf{to} $\texttt{n} - 1$ \textbf{step} $\texttt{r}$}
        \STATE{ \textbf{yield} $i$ }
    \ENDFOR
    \STATE{ $\texttt{last\_rank} \gets \texttt{n} - 1$ }
    \IF{$\texttt{last\_rank} > 0$ \textbf{and} $(\texttt{last\_rank} \mod \texttt{r}) \neq 0$}
        \STATE{ \textbf{yield} $\texttt{last\_rank}$ }
    \ENDIF
\end{algorithmic}
\end{algorithm}
