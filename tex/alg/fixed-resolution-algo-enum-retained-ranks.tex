\begin{algorithm}
\caption{Fixed Resolution Stratum Enumeration}
\label{alg:fixed-resolution-algo-enum-retained-rans}
\begin{algorithmic}[1]
    \Require{$n \in \mathbb{N}_{\geq0}$ -- the number of strata deposited}
    \Require{$r \in \mathbb{N}_{>0}$ -- the fixed resolution desired}
    \For{$i = 0$ \textbf{to} $\texttt{n} - 1$ \textbf{step} $\texttt{r}$}
        \State{ \textbf{yield} $i$ }
    \EndFor
    \If{$n > 0$ and $n - 1 \bmod r > 0$}
        \State{ \textbf{yield} $n - 1$ }
    \EndIf
\end{algorithmic}
\end{algorithm}
